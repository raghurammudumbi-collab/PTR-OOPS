# Demonstrating Overloading and Overriding

class Calculator:
    # Python doesn't support true overloading; we simulate it with default arguments
    def add(self, a, b=0, c=0):
        return a + b + c

class AdvancedCalculator(Calculator):
    # Method Overriding: redefining the same method
    def add(self, a, b=0, c=0):
        print("Adding numbers in AdvancedCalculator")
        return super().add(a, b, c) + 10  # modified behavior

# Usage
calc = Calculator()
print(calc.add(2, 3))        # Simulated overloading
print(calc.add(2, 3, 4))     # Simulated overloading

adv_calc = AdvancedCalculator()
print(adv_calc.add(2, 3))    # Overridden method
